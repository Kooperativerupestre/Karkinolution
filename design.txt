    @staticmethod
    def score_atack(creature:Entity, target:Entity) -> float:
        courage = COURAGE_FACTOR[creature.genome.behavior]
        trade_off_base = TRADE_OFF[creature.genome.behavior]
        difference = creature.physical_ratio/(creature.physical_ratio + target.physical_ratio)
        
        fitness = LimitedValue(1, 1)

        if creature.gender is Gender.FEMALE:
            if creature.uterus.pregnant: # type: ignore
                fitness.sub(creature.uterus.pregnancy_factor)# type: ignore

        factor = courage*1.1 * trade_off_base * difference * fitness

        return factor
